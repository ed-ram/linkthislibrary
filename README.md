## Install

```bash
npm install --save linkthislibrary
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'linkthislibrary'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
